### 代码：

```js
// 将一个长方形横放改为竖放 每列为一个数组
var transpose = function(A) {
  let result = [];
  // 横向遍历 长度
  for (let i = 0; i < A[0].length; i++) {
    let row = [];
    // 纵向遍历 宽度
    for (let j = 0; j < A.length; j++) {
      // 添加宽度的值
      row.push(A[j][i]);
    }
    result.push(row);
  }
  return result;
};
```
